28 inline std::vector<int> levit_algorithm(int n, const std::vector<std::vector<std::pair<int, int>>>& graph, int start) {
29  std::vector<int> dist(n, INF);
30  dist[start] = 0;
31 
32  std::vector<int> m(n, 2);
33  m[start] = 1;
34 
35  std::deque<int> q;
36  q.push_back(start);
37 
38  while (!q.empty()) {
39  int v = q.front();
40  q.pop_front();
41  m[v] = 0;
42 
43  for (const auto& [u, w] : graph[v]) {
44  if (dist[u] > dist[v] + w) {
45  dist[u] = dist[v] + w;
46 
47  if (m[u] == 2) {
48  m[u] = 1;
49  q.push_back(u);
50  } else if (m[u] == 0) {
51  m[u] = 1;
52  q.push_front(u);
53  }
54  }
55  }
56  }
57 
58  return dist;
59 }

68 inline std::pair<json, std::vector<std::vector<std::pair<int, int>>>> generate_random_graph(int n, int m, int max_weight) {

71  std::uniform_int_distribution<> weight_dist(1, max_weight);
72  std::uniform_int_distribution<> node_dist(0, n-1);
73 
74  std::vector<std::vector<std::pair<int, int>>> graph(n);

76 
77  for (int i = 0; i < m; ++i) {
78  int u = node_dist(gen);
